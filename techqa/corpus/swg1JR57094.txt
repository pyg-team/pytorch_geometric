LOCAL FIX
 *  potential workaround:
    Check for any JMX application running on your system and
    disable it. Afterwards restart your environment.
   
   workaround:
    Write a JMX client Creating an AdminClient to the AppTarget
    Server. Install it on the AppTarget and execute regularly
    to avoid the NullConnector is used.
   ----------------------------------------------------------------
   Properties connectProps = new Properties();
   connectProps.setProperty(AdminClient.CONNECTOR_TYPE,
     AdminClient. CONNECTOR_TYPE_SOAP);
   connectProps.setProperty(AdminClient.CONNECTOR_HOST,
     "<YOUR_HOST NAME>");
   connectProps.setProperty(AdminClient.CONNECTOR_PORT,
     "<SOAP_PORT>");
   connectProps.setProperty(AdminClient.USERNAME,
     "<AUTHORIZED_USER>");
   connectProps.setProperty(AdminClient.PASSWORD, "<PASSWORD>");
   
   // might be required in your environment
   // System.setProperty("com.ibm.SSL.ConfigURL",
   //"<YOUR_PROFILE_DIR>/properties/ssl.client.props");
   
   AdminClient adminClient = AdminClientFactory
          .createAdminClient(connectProps);
   
   CommandMgr cmdmgr = CommandMgr.getCommandMgr(adminClient);
