 * Only one table's indexes:
   
   COLUMN INDEX_NAME FORMAT A20
   COLUMN TABLE_NAME FORMAT A20
   COLUMN COLUMN_NAME FORMAT A15
   COLUMN UNIQUENES FORMAT A10
   COLUMN COLUMN_NAME FORMAT A15
   SELECT a.TABLE_NAME,a.INDEX_NAME,a.UNIQUENESS,b.COLUMN_NAME,b.COLUMN_POSITION FROM ALL_INDEXES a
   JOIN ALL_IND_COLUMNS b ON a.INDEX_NAME=b.INDEX_NAME
   WHERE a.TABLE_NAME = '<table_name>' ORDER BY a.TABLE_NAME,a.INDEX_NAME,b.COLUMN_POSITION;



MySQL: 

 * All Build Forge indexes:
   
   SELECT TABLE_NAME,INDEX_NAME,COLUMN_NAME,NON_UNIQUE,INDEX_TYPE
   from information_schema.statistics 
   WHERE INDEX_NAME LIKE 'bf_%' ORDER BY TABLE_NAME,INDEX_NAME,COLUMN_NAME;
   

 * Only one table's indexes:
   
   SELECT TABLE_NAME,INDEX_NAME,COLUMN_NAME,NON_UNIQUE,INDEX_TYPE
   from information_schema.statistics
   WHERE TABLE_NAME='<table_name>' ORDER BY TABLE_NAME,INDEX_NAME,COLUMN_NAME;
   



DB2: 

 * All Build Forge indexes:
   
   SELECT INDNAME,COLNAMES, UNIQUERULE FROM SYSCAT.INDEXES WHERE INDNAME LIKE 'BF_%'

 * Only one table's indexes:
   
   SELECT INDNAME,COLNAMES, UNIQUERULE FROM SYSCAT.INDEXES WHERE INDNAME LIKE 'BF_%' AND TABNAME = 'TABLE_NAME'
