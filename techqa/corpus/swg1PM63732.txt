Title: IBM PM63732: JAVASCRIPT ERROR IN MSIE8 WHEN EDITING CONTENT - United States

Text:
  A FIX IS AVAILABLE
Fixes integrated in WebSphere Portal 7.0.0.1 & 7.0.0.2 Combined Cumulative Fixes [http://www-01.ibm.com/support/docview.wss?uid=swg24029452]



APAR STATUS
 * CLOSED AS PROGRAM ERROR.
    
   
   

ERROR DESCRIPTION
 *  When editing a page with the WCM Advanced Editor using an IE
   browser and the PageBuilder2 theme, you will receive an error:
   "Tried to register widget with id==ibmStatusBarContainer but
   that id is already registered."
   
   This APAR modifies the PageBuilder2 theme to fix the problem.
   Opened for release 7001
   
   
    
   
   

LOCAL FIX
 *  If the customer wishes to patch their theme with this fix, they
   can do the following.
   
   Connect to WebDav http://server:port/wps/mycontenthandler/dav/fs [http://server:port/wps/mycontenthandler/dav/fs]
   -type1/, then navigate to themes\PageBuilder2\nls.  Edit the
   Theme_*.html that matches the locale(s) you are using (for me it
   is theme_en.html).
   
   Find the first script element which looks like this:
   
   &lt;script type="text/javascript"&gt;
   (function(){
       var commonInitOnload = function(){
       com.ibm.pb.themes.commonInit({
       setWindowTitle:false, // title is set by the head dynamic
   content spot
       useNavigationController: true,
       useRenderingController: true,
       useDNDController: true,
       initLiveTextService: true,
       lazyLoadModeWidgets: true,
       navPrimingContainers:
       ["selectionPathPrimer","topNavLinks","navTabsRoot"],
       customInit: com.ibm.themes.PageBuilder2.init
       });
       dojo.publish("com.ibm.portal.theme.portlet_ready"); //
   notifies ASA that portlet IDs are ready to be found in the DOM
   }
   if (dojo.isIE) {
       dojo.connect('onload', commonInitOnload);
   } else {
       dojo.addOnLoad(commonInitOnload);
   }
   })();
   &lt;/script&gt;
   
   and replace it with this:
   
   &lt;script type="text/javascript"&gt;
   (function(){
       var commonInitOnload = function(){
       com.ibm.pb.themes.commonInit({
       setWindowTitle:false, // title is set by the head dynamic
   content spot
       useNavigationController: true,
       useRenderingController: true,
       useDNDController: true,
       initLiveTextService: true,
       lazyLoadModeWidgets: true,
       navPrimingContainers:
       ["selectionPathPrimer","topNavLinks","navTabsRoot"],
       customInit: com.ibm.themes.PageBuilder2.init
       });
       if (dojo.isIE) {
           dojo.disconnect(ie_commonInitOnload);
       }
       dojo.publish("com.ibm.portal.theme.portlet_ready"); //
   notifies ASA that portlet IDs are ready to be found in the DOM
   }
   if (dojo.isIE) {
       var ie_commonInitOnload = dojo.connect(window, 'onload',
   commonInitOnload);
   } else {
       dojo.addOnLoad(commonInitOnload);
   }
   })();
   &lt;/script&gt;
   
   This would have to be repeated for all locales which the theme
   uses.
