ERROR DESCRIPTION
 *  Confirmation dialog that is thrown by Internet Explorer when we
   try to close the active window programmatically from
   tm_process_finished.jsp.
   
   
    
   
   

LOCAL FIX
 *  Workaround is replacing self.close() call with
   browser-dependent code that performs one of specific technics
   to close a window - see the code at the bottom.
   
   if (navigator.appName.indexOf('Explorer') > 0) {
     var version = 0;
     if (/MSIE\s*(\d+)/.test(navigator.userAgent)) {
       version = Number(RegExp.$1);
     }
     if (version >= 7) {
       window.open('', '_self', '');
     }
     else if (version == 6) {
       window.opener = null;
     }
     else {
       window.opener = '';
     }
   }
   else if (/^Mozilla/.test(navigator.userAgent)) {
     console.error("window.close() will not work in Firefox");
     return;
   }
   window.close();
